Como utilizar:


import Mp3Lite;
import Mp3Event;
var myplayer:Mp3Lite = new Mp3Lite();
myplayer.setXMLList("playlist.xml");
myplayer.addEventListener(Mp3Event.ON_START, function(e) {trace("Let's rock!"});
myplayer.start();
